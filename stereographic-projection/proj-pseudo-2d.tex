\documentclass{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.18}

\begin{document}
\begin{tikzpicture}
\begin{axis}[
    axis lines=middle,
    xlabel=$x$,
    ylabel=$y$,
    xmin=-3, xmax=3,
    ymin=-2.5, ymax=2.5,
    ticks=none,
    grid=none,
    clip=false,
    axis equal image,
    width=8cm,
    height=8cm,
]

% Hyperbola x² - y² = 1 (equivalent to -x² + y² = -1)
\addplot [domain=1:3, samples=100, blue] {sqrt(x^2 - 1)};
\addplot [domain=1:3, samples=100, blue] {-sqrt(x^2 - 1)};
% \addplot [domain=-3:-1, samples=100, blue] {sqrt(x^2 - 1)};
% \addplot [domain=-3:-1, samples=100, blue] {-sqrt(x^2 - 1)};
% \node[blue, right] at (axis cs: 2, 1.5) {$-x^2 + y^2 = -1$};

% Point S at (-1, 0)
\node[circle, fill=black, inner sep=1.5pt, label=below left:$S$] at (axis cs: -1, 0) {};

% Point P at (2, sqrt(3))
\pgfmathsetmacro{\px}{2}
\pgfmathsetmacro{\py}{sqrt(3)}
\coordinate (P) at (axis cs: \px, \py);
\node[circle, fill=black, inner sep=1.5pt, label=above:$P$] at (P) {};

\node[circle, fill=black, inner sep=1.5pt, label=below left:$O$] at (0, 0) {};
\node[circle, fill=black, inner sep=1.5pt, label=below left:$O'$] at (\px, 0) {};

% Line SP
\draw [red] (axis cs: -1, 0) -- (P);

% Intersection σ(P) at (0, sqrt(3)/3)
\pgfmathsetmacro{\sigmay}{\py / (\px + 1)}
\coordinate (sigma) at (axis cs: 0, \sigmay);
\node[circle, fill=black, inner sep=1.5pt, label=left:$\sigma(P)$] at (sigma) {};

\draw [red] (sigma) -- (0, 0);
\draw [red] (P) -- (\px, 0);
\end{axis}
\end{tikzpicture}
\end{document}
